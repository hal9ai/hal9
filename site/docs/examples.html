Charts

Bar

  • R
  • Python
library(hal9)

h9_create() |>
  h9_load(mtcars) |>
  h9_bar_chart(x = "cyl",
               y = "carb",
               type = "grouped",
               orientation = "horizontal",
               palette = "schemePaired",
               fontsize = "14",
               tickrotation = "0",
               marginleft = "40",
               marginbottom = "30")
"soon"
